✨ КОМАНДЫ УПРАВЛЕНИЯ:

PowerShell:
  .\run.ps1 -Command docker-up      # Запустить Docker
  .\run.ps1 -Command docker-down    # Остановить Docker
  .\run.ps1 -Command logs           # Логи backend
  .\run.ps1 -Command backend        # Backend локально
  .\run.ps1 -Command frontend       # Frontend локально
  .\run.ps1 -Command install        # Установить зависимости
  .\run.ps1 -Command migrate        # Миграции БД
  .\run.ps1 -Command clean          # Очистить контейнеры
